ruleValidate(){
	if(this.organDialogEditLevel!=0){
	    return {
		name: [{required: true, message: '请输入', trigger: 'blur' ,}],
		organ:[{
		    required: true, message: '请下拉选择', trigger: 'blur,change' ,
		    validator:(rule,value,callback)=>{
			if(this.organDialogForm.organ.name){
			    callback()
			}else{
			    callback(new Error());
			}
		    } 
		}]
	    }
	}else{
	    return {
		name: [{required: true, message: '请输入', trigger: 'blur' ,}],
	    }
	}
}
